# 环境部署
## MySQL安装
### 安装依赖包
```
yum install -y libaio*
```

### 创建mysql账号
```
groupadd mysql
useradd -r -g mysql -m mysql
```

### 解压mysql安装包
```
cd /data/software
tar -xzvf mysql-5.7.24-linux-glibc2.12-x86_64.tar.gz
mv mysql-5.7.24-linux-glibc2.12-x86_64 /data/
cd /data/
ln -s mysql-5.7.24-linux-glibc2.12-x86_64 mysql
```

### 修改目录属性
```
chown -R mysql.mysql  /data/mysql-5.7.24-linux-glibc2.12-x86_64
```

### 初始化mysql
```
su - mysql
cd /data/mysql
mkdir -p data
mkdir logs
mkdir data/binlog
touch logs/mysql.log
chown -R mysql.mysql .

bin/mysqld --initialize --user=mysql --basedir=/data/mysql --datadir=/data/mysql/data
```

### 修改目录权限
```
su - root
cd /data/mysql
chown -R root .  
chown -R mysql data
chown -R mysql logs
```

### 配置my.cnf文件
```
cat > /etc/my.cnf
[client]
port = 3306
socket = /data/mysql/data/mysql.sock

# MySQL server
[mysqld]
port             = 3306
user             = mysql
socket           = /data/mysql/data/mysql.sock
pid-file         = /data/mysql/data/mysql.pid
basedir          = /data/mysql
datadir          = /data/mysql/data
tmpdir           = /tmp
open_files_limit = 10240
sql_mode         = NO_ENGINE_SUBSTITUTION,STRICT_TRANS_TABLES
skip-host-cache
skip-name-resolve
skip-external-locking
explicit_defaults_for_timestamp

# Buffer Tuning
max_allowed_packet       = 256M
max_heap_table_size      = 256M
net_buffer_length        = 8K
sort_buffer_size         = 2M
join_buffer_size         = 4M
read_buffer_size         = 2M
read_rnd_buffer_size     = 16M
thread_stack             = 192K
thread_cache_size        = 8
max_connections          = 100

# Query Cache Configuration
query_cache_limit        = 1M
query_cache_size         = 16M

character_set_server     = utf8
character_set_filesystem = utf8
collation-server         = utf8_general_ci
init-connect             = 'SET NAMES utf8'
init_connect             = 'SET collation_connection = utf8_general_ci'
skip-character-set-client-handshake

# Log
log-error             = /data/mysql/logs/mysql.log
log_bin               = /data/mysql/data/binlog/mysql-bin.log
server_id             = 100
binlog_cache_size     = 32m
max_binlog_cache_size = 512m
max_binlog_size       = 512m
binlog_format         = mixed
log_output            = FILE

slow_query_log = 1
slow_query_log_file = ../slow_query.log
long_query_time  =2
general_log = 0
general_log_file = ../general_query.log
expire-logs-days = 14

# InnoDB
innodb_log_file_size      = 256M
innodb_log_files_in_group = 3
innodb_buffer_pool_size   = 1024M
innodb_log_buffer_size    = 8M
innodb_file_per_table     = 1
innodb_open_files         = 400
innodb_io_capacity        = 1000
innodb_flush_method       = O_DIRECT

[mysql]
no-auto_rehash
prompt = (\u@\h) [\d]>\_
default-character-set = gbk
```

### 将mysqld添加至主机服务
```
chown mysql:mysql /etc/my.cnf

cp support-files/mysql.server /etc/init.d/mysqld

chmod +x /etc/init.d/mysqld

vi /etc/init.d/mysqld
1、修改填充basedir和datadir
2、在266行添加--defaults-file（不加的话会报错找不到pid文件）
$bindir/mysqld_safe --defaults-file=/etc/my.cnf --datadir="$datadir" --pid-file="$mysqld_pid_file_path" $other_args >/dev/null &

chkconfig --add mysqld
chkconfig --list mysqld

```

### 启动MySQL
```
service mysqld start

# 修改mysql默认root密码-略
```

## Nginx部署
### 安装Nginx
```
wget -O /etc/yum.repos.d/epel-7.repo http://mirrors.aliyun.com/repo/epel-7.repo 

yum install -y nginx
```

### 配置Nginx
```
mv /etc/nginx/nginx.conf /etc/nginx/nginx.conf.bak

cat > /etc/nginx/nginx.conf
user  nginx;
worker_processes  1;

error_log  /var/log/nginx/error.log warn;
pid        /var/run/nginx.pid;

events {
    worker_connections  1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;

    sendfile        on;
    #tcp_nopush     on;

    keepalive_timeout  65;

    #gzip  on;

    include /etc/nginx/conf.d/*.conf;

    upstream dataquality.utrustfintech.com {
        server localhost:9000;
    }

    # gunicorn
    server {
        listen 80;
        server_name dataquality.utrustfintech.com;
        location / {
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_pass http://dataquality.utrustfintech.com;
        }

        location /static {
                alias /data/data-quality/static;
        }
    }
}
```

# 防火墙设置
```
service iptables stop

# 查看Firewalld状态
firewall-cmd --state

# 启动firewalld服务
systemctl start firewalld

# 把服务加入开机自启
systemctl enable firewalld.service

# 放行端口号
firewall-cmd --zone=public --add-port=80/tcp --permanent
firewall-cmd --zone=public --add-port=3306/tcp --permanent

# 重载防火墙配置
firewall-cmd --reload

# 查看已放行端口
firewall-cmd --zone=public --list-ports

# 禁用SELinux
vim /etc/selinux/config
SELINUX=permissive
```
